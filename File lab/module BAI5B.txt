module BAI5B(
output [3:0] Q,
input clk,
input rst,
input up_down);
reg [3:0] value;
always @(posedge clk)
begin
if(-rst)
value <= 4'h0;
else if (up_down)
value <= value + 4'dl;
else
value <= value - 4'dl;
end
assign Q = value;
endmodule

module BAI5B_test(
output [3:0] LEDR,
output [2:0] LEDG,
input [1:0] KEY,
input [0:0] SW);
assign LEDG [1:0] = KEY [1:0];
assign LEDG[2] = SW[0];
BAI5B DUT (
.Q(LEDR [3:0]), .clk(KEY [1]), .rst (KEY[0]), .up_down (SW[0]));
endmodule
