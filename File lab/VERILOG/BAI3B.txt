module BAI3B(
output wire SUM,
output wire CARRY,
input wire A,
input wire B,
input wire C);

wire HALF_SUM1, HALF_CARRY1, HALF_CARRY2;

BAI3A HA1(HALF_SUM1, HALF_CARRY1, A, B);
BAI3A HA2(SUM, HALF_CARRY2, HALF_SUM1, C);
assign CARRY = HALF_CARRY1 | HALF_CARRY2;
endmodule

BAI3B DUT(
