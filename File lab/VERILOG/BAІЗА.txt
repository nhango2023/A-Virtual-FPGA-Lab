module BAI3A (
output wire SUM,
output wire CARRY,
input wire A,
input wire B);

assign SUM = A ^ B;
assign CARRY = A & B;
endmodule

module BAI3A_test(
output [1:0] LEDR, 
output [1:0] LEDG,
input [1:0]SW);

assign LEDR[1:0] = SW[1:0];

BAI3A DUT(
.SUM(LEDG[1]), 
CARRY(LEDG[0]),
.A(SW[0]),
.B(SW[1]));
endmodule
